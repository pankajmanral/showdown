Get the customer's order details 


c.user_id -> admin panel -> customer model displayed id  
SELECT
    o.order_uuid,
    o.order_on,
    o.status AS order_status,
    o.order_time as order_time,
    od.quantity AS product_quantity,
    od.price AS product_price,
    p.product_name,
    p.product_description,
    u.first_name,
    u.last_name,
    a.title AS address_title,
    a.block_number,
    a.building,
    a.street,
    a.land_mark,
    a.area,
    a.city,
    a.pincode,
    a.state
FROM
    `order_order` o
JOIN
    `order_orderdetails` od ON o.id = od.order_id
JOIN
    `product` p ON od.product_id = p.id
JOIN
    `account_customer` c ON o.customer_id = c.id
JOIN
    `auth_user` u ON c.user_id = u.id  -- Join User to get first_name and last_name
JOIN
    `account_address` a ON o.shipping_address_id = a.id  -- Join Address to get address details
WHERE
    c.user_id = cust_id
ORDER BY
    o.order_on DESC;
